  function calificar(data) {

    let puntos = 0;

    let motivos = [];


    // Edad

    if (data.edad >= 23 && data.edad <= 45) puntos += 1;

    else motivos.push("edad fuera de rango preferente (23-45)");


    // Estudios

    if (data.estudios.includes("Licenciatura")) puntos += 2;

    else if (data.estudios.includes("Pasante")) puntos += 1;

    else motivos.push("nivel de estudios no relacionado");


    // Experiencia (mínimo 1 año)

    if (data.experiencia >= 1) puntos += 2;

    else motivos.push("menos de 1 año de experiencia");


    // AutoCAD

    if (data.autocad === "Sí, con experiencia") puntos += 1;

    else if (data.autocad === "No") motivos.push("sin manejo de AutoCAD");


    // Office

    if (data.office === "Sí, con experiencia") puntos += 1;

    else if (data.office === "No") motivos.push("sin manejo de Office");


    // Procesos constructivos

    if (data.procesos === "Sí") puntos += 1;

    else if (data.procesos === "No") motivos.push("sin conocimiento de procesos constructivos");


    // Horario — eliminatorio

    if (data.horario === "No") {

      return { calificacion: "NO CUMPLE (horario)", puntos, motivos: ["no puede cumplir el horario requerido"] };

    }


    if (puntos >= 6) return { calificacion: "CUMPLE REQUISITOS", puntos, motivos };

    if (puntos >= 4) return { calificacion: "REVISAR (parcial)", puntos, motivos };

    return { calificacion: "NO CUMPLE", puntos, motivos };

  }

    const data = {

      institucion: document.getElementById("institucion").value,

      experiencia: parseFloat(document.getElementById("experiencia").value),

      experienciaResumen: document.getElementById("experienciaResumen").value,

      autocad: document.querySelector('input[name="autocad"]:checked').value,

      office: document.querySelector('input[name="office"]:checked').value,

      procesos: document.querySelector('input[name="procesos"]:checked').value,

      otrosProgramas: document.getElementById("otrosProgramas").value,

      horario: document.querySelector('input[name="horario"]:checked').value,

    };


    const eval_ = calificar(data);

    data.calificacion = eval_.calificacion + " (" + eval_.puntos + " pts)";

        await fetch(SCRIPT_URL, {

      }

Tus datos quedaron registrados por si surge una posición más adecuada.</p>";
